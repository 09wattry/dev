def count_startswith(L, ch):
    """ (list of str, str) -> int
    Precondition: the length of each item in L is >= 
        1, and len(ch) == 1
    Return the number of strings in L that begin 
        with ch.
  
    >>> count_startswith(['rumba', 'salsa', 'samba']
        , 's')
    2
    """
    ch_strings = []
    for item in L:
        if item[0] == ch:
            ch_strings.append(item)
    return len(ch_strings)

def is_one_to_one(d):
    """ (dict) -> bool
 
    Return True if and only if no two of d's keys 
        map to the same value.
    >>> is_one_to_one({'a': 1, 'b': 2, 'c': 3})
    True
    >>> is_one_to_one({'a': 1, 'b': 2, 'c': 1})
    False
    >>> is_one_to_one({})
    True
    """
    seen = []  # The values that have been seen so 
        far.
    for k in d:
        if d[k] in seen:
             return False
        else:
            seen.append(d[k])
    return True
